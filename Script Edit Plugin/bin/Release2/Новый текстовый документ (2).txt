Script #addons
(
  if
  (
    isFileExists(StrCat("maps\","Addons\","addon_",FloatToStr(addon),".mob"))
    isAlive(GetLeader())
    isLess(addon,20)
    isEqual(GsGetVar(0,"ProgressCheck"),1)
  )
  then
  (
    KillScript()
    AddMob(StrCat("Addons\","addon_",FloatToStr(addon),".mob"))
    SleepUntil(Not( isEqualString(LoadedAddon,"") ))
    ConsoleString( StrCat( "[Addon] ", FloatToStr(addon), ": ", LoadedAddon ) )
    LoadedAddon = ""
    #addons(Add(addon,1))
  )
  if
  (
    isAlive(GetLeader())
    isLess(addon,20)
    isEqual(GsGetVar(0,"ProgressCheck"),1)
  )
  then
  (
    KillScript()
    #addons(Add(addon,1))
  )
  if
  (
    isAlive(GetLeader())
    isEqual(GsGetVar(0,"ProgressCheck"),1)
  )
  then
  (
    KillScript()
    HideObject(GetObjectByID("1000059186"),0)//Doska
  )
)

Script #buyMerc
(
  if
  (
    IsEqual(GSGetPlayerVar(0,StrCat("gvar.",this)),0)
    IsEqual(GSGetVar(0,StrCat(posrednik,this)),0)
    isGreater(progressNow,progressMin)
    isLess(progressNow,progressMax)
    isALive(GetLeader())
  )
  then
  (
    KillScript()
    GSSetVar(0,StrCat(posrednik,this),1)
    #buyMerc(posrednik,this,prise,progressMin,progressMax)
  )
  if
  (
    IsEqual(GSGetPlayerVar(0,StrCat("gvar.",this)),0)
    IsEqual(GSGetVar(0,StrCat(posrednik,this)),2)
    isGreater(progressNow,progressMin)
    isLess(progressNow,progressMax)
    isALive(GetLeader())
    isGreater(Sub(GetMoney( GetPlayer(GetLeader()) ), prise) ,0)
  )
  then
  (
    KillScript()
    GiveMoney(GetPlayer(GetLeader()),Mul(prise,-1))
    GsSetPlayerVar(0,StrCat("gvar.",this),1)
    SendCurrentParty(0)
    Say(StrCat("Paid: ",FloatToStr(prise)))
    #buyMerc(posrednik,this,prise,progressMin,progressMax)
  )
